    84→def label_smoothing(y: np.ndarray, alpha=0.1):
    85→    """
    86→    Applies label smoothing to the given labels.
    87→    Label smoothing is a technique used to prevent the model from becoming overconfident by adjusting the target labels.
    88→    It subtracts a small value (alpha) from the correct label and distributes it among the other labels.
    89→    Args:
    90→        y (numpy.ndarray): Array of labels to be smoothed. The array should be of shape (num_labels,).
    91→        alpha (float, optional): Smoothing parameter. Default is 0.1.
    92→    Returns:
    93→        numpy.ndarray: The smoothed labels.
    94→    """
    95→    # Subtract alpha from correct label when it is >0
    96→    y[y > 0] -= alpha
    97→
    98→    # Assigned alpha to all other labels
    99→    y[y == 0] = alpha / y.shape[0]
   100→
   101→    return y

   104→def mixup(x, y, augmentation_ratio=0.25, alpha=0.2):
   105→    """Apply mixup to the given data.
   106→
   107→    Mixup is a data augmentation technique that generates new samples by
   108→    mixing two samples and their labels.
   109→
   110→    Args:
   111→        x: Samples.
   112→        y: One-hot labels.
   113→        augmentation_ratio: The ratio of augmented samples.
   114→        alpha: The beta distribution parameter.
   115→
   116→    Returns:
   117→        Augmented data.
   118→    """
   119→    rng = np.random.default_rng(cfg.RANDOM_SEED)
   120→
   121→    # Get indices of all positive samples
   122→    positive_indices = np.unique(np.where(y[:, :] == 1)[0])
   123→
   124→    # Calculate the number of samples to augment based on the ratio
   125→    num_samples_to_augment = int(len(positive_indices) * augmentation_ratio)
   126→
   127→    # Indices of samples, that are already mixed up
   128→    mixed_up_indices = []
   129→
   130→    for _ in range(num_samples_to_augment):
   131→        # Randomly choose one instance from the positive samples
   132→        index = rng.choice(positive_indices)
   133→
   134→        # Choose another one, when the chosen one was already mixed up
   135→        while index in mixed_up_indices:
   136→            index = rng.choice(positive_indices)
   137→
   138→        x1, y1 = x[index], y[index]
   139→
   140→        # Randomly choose a different instance from the dataset
   141→        second_index = rng.choice(positive_indices)
   142→
   143→        # Choose again, when the same or an already mixed up sample was selected
   144→        while second_index == index or second_index in mixed_up_indices:
   145→            second_index = rng.choice(positive_indices)
   146→        x2, y2 = x[second_index], y[second_index]
   147→
   148→        # Generate a random mixing coefficient (lambda)
   149→        lambda_ = rng.beta(alpha, alpha)
   150→
   151→        # Mix the embeddings and labels
   152→        mixed_x = lambda_ * x1 + (1 - lambda_) * x2
   153→        mixed_y = lambda_ * y1 + (1 - lambda_) * y2
   154→
   155→        # Replace one of the original samples and labels with the augmented sample and labels
   156→        x[index] = mixed_x
   157→        y[index] = mixed_y
   158→
   159→        # Mark the sample as already mixed up
   160→        mixed_up_indices.append(index)
   161→
   162→        del mixed_x
   163→        del mixed_y
   164→
   165→    return x, y

   168→def random_split(x, y, val_ratio=0.2):
   169→    """Splits the data into training and validation data.
   170→
   171→    Makes sure that each class is represented in both sets.
   172→
   173→    Args:
   174→        x: Samples.
   175→        y: One-hot labels.
   176→        val_ratio: The ratio of validation data.
   177→
   178→    Returns:
   179→        A tuple of (x_train, y_train, x_val, y_val).
   180→    """
   181→    rng = np.random.default_rng(cfg.RANDOM_SEED)
   182→
   183→    # Get number of classes
   184→    num_classes = y.shape[1]
   185→
   186→    # Initialize training and validation data
   187→    x_train, y_train, x_val, y_val = [], [], [], []
   188→
   189→    # Split data
   190→    for i in range(num_classes):
   191→        # Get indices of positive samples of current class
   192→        positive_indices = np.where(y[:, i] == 1)[0]
   193→
   194→        # Get indices of negative samples of current class
   195→        negative_indices = np.where(y[:, i] == -1)[0]
   196→
   197→        # Get number of samples for each set
   198→        num_samples = len(positive_indices)
   199→        num_samples_train = max(1, int(num_samples * (1 - val_ratio)))
   200→        num_samples_val = max(0, num_samples - num_samples_train)
   201→
   202→        # Randomly choose samples for training and validation
   203→        rng.shuffle(positive_indices)
   204→        train_indices = positive_indices[:num_samples_train]
   205→        val_indices = positive_indices[num_samples_train : num_samples_train + num_samples_val]
   206→
   207→        # Append samples to training and validation data
   208→        x_train.append(x[train_indices])
   209→        y_train.append(y[train_indices])
   210→        x_val.append(x[val_indices])
   211→        y_val.append(y[val_indices])
   212→
   213→        # Append negative samples to training data
   214→        x_train.append(x[negative_indices])
   215→        y_train.append(y[negative_indices])
   216→
   217→    # Add samples for non-event classes to training and validation data
   218→    non_event_indices = np.where(np.sum(y[:, :], axis=1) == 0)[0]
   219→    num_samples = len(non_event_indices)
   220→    num_samples_train = max(1, int(num_samples * (1 - val_ratio)))
   221→    num_samples_val = max(0, num_samples - num_samples_train)
   222→    rng.shuffle(non_event_indices)
   223→    train_indices = non_event_indices[:num_samples_train]
   224→    val_indices = non_event_indices[num_samples_train : num_samples_train + num_samples_val]
   225→    x_train.append(x[train_indices])
   226→    y_train.append(y[train_indices])
   227→    x_val.append(x[val_indices])
   228→    y_val.append(y[val_indices])
   229→
   230→    # Concatenate data
   231→    x_train = np.concatenate(x_train)
   232→    y_train = np.concatenate(y_train)
   233→    x_val = np.concatenate(x_val)
   234→    y_val = np.concatenate(y_val)
   235→
   236→    # Shuffle data
   237→    indices = np.arange(len(x_train))
   238→    rng.shuffle(indices)
   239→    x_train = x_train[indices]
   240→    y_train = y_train[indices]
   241→
   242→    indices = np.arange(len(x_val))
   243→    rng.shuffle(indices)
   244→    x_val = x_val[indices]
   245→    y_val = y_val[indices]
   246→
   247→    return x_train, y_train, x_val, y_val

   250→def random_multilabel_split(x, y, val_ratio=0.2):
   251→    """Splits the data into training and validation data.
   252→
   253→    Makes sure that each combination of classes is represented in both sets.
   254→
   255→    Args:
   256→        x: Samples.
   257→        y: One-hot labels.
   258→        val_ratio: The ratio of validation data.
   259→
   260→    Returns:
   261→        A tuple of (x_train, y_train, x_val, y_val).
   262→
   263→    """
   264→    rng = np.random.default_rng(cfg.RANDOM_SEED)
   265→
   266→    # Find all combinations of labels
   267→    class_combinations = np.unique(y, axis=0)
   268→
   269→    # Initialize training and validation data
   270→    x_train, y_train, x_val, y_val = [], [], [], []
   271→
   272→    # Split the data for each combination of labels
   273→    for class_combination in class_combinations:
   274→        # find all indices
   275→        indices = np.where((y == class_combination).all(axis=1))[0]
   276→
   277→        # When negative sample use only for training
   278→        if -1 in class_combination:
   279→            x_train.append(x[indices])
   280→            y_train.append(y[indices])
   281→        # Otherwise split according to the validation split
   282→        else:
   283→            # Get number of samples for each set
   284→            num_samples = len(indices)
   285→            num_samples_train = max(1, int(num_samples * (1 - val_ratio)))
   286→            num_samples_val = max(0, num_samples - num_samples_train)
   287→            # Randomly choose samples for training and validation
   288→            rng.shuffle(indices)
   289→            train_indices = indices[:num_samples_train]
   290→            val_indices = indices[num_samples_train : num_samples_train + num_samples_val]
   291→            # Append samples to training and validation data
   292→            x_train.append(x[train_indices])
   293→            y_train.append(y[train_indices])
   294→            x_val.append(x[val_indices])
   295→            y_val.append(y[val_indices])
   296→
   297→    # Concatenate data
   298→    x_train = np.concatenate(x_train)
   299→    y_train = np.concatenate(y_train)
   300→    x_val = np.concatenate(x_val)
   301→    y_val = np.concatenate(y_val)
   302→
   303→    # Shuffle data
   304→    indices = np.arange(len(x_train))
   305→    rng.shuffle(indices)
   306→    x_train = x_train[indices]
   307→    y_train = y_train[indices]
   308→
   309→    indices = np.arange(len(x_val))
   310→    rng.shuffle(indices)
   311→    x_val = x_val[indices]
   312→    y_val = y_val[indices]
   313→
   314→    return x_train, y_train, x_val, y_val

   317→def upsample_core(x: np.ndarray, y: np.ndarray, min_samples: int, apply, size=2):
   318→    """
   319→    Upsamples the minority class in the dataset using the specified apply function.
   320→    Parameters:
   321→        x (np.ndarray): The feature matrix.
   322→        y (np.ndarray): The target labels.
   323→        min_samples (int): The minimum number of samples required for the minority class.
   324→        apply (callable): A function that applies the SMOTE or any other algorithm to the data.
   325→        size (int, optional): The number of samples to generate in each iteration. Default is 2.
   326→    Returns:
   327→        tuple: A tuple containing the upsampled feature matrix and target labels.
   328→    """
   329→    rng = np.random.default_rng(cfg.RANDOM_SEED)
   330→    y_temp = []
   331→    x_temp = []
   332→
   333→    if cfg.BINARY_CLASSIFICATION:
   334→        # Determine if 1 or 0 is the minority class
   335→        minority_label = 1 if y.sum(axis=0) < len(y) - y.sum(axis=0) else 0
   336→
   337→        while np.where(y == minority_label)[0].shape[0] + len(y_temp) < min_samples:
   338→            # Randomly choose a sample from the minority class
   339→            random_index = rng.choice(np.where(y == minority_label)[0], size=size)
   340→
   341→            # Apply SMOTE
   342→            x_app, y_app = apply(x, y, random_index)
   343→            y_temp.append(y_app)
   344→            x_temp.append(x_app)
   345→    else:
   346→        for i in range(y.shape[1]):
   347→            while y[:, i].sum() + len(y_temp) < min_samples:
   348→                try:
   349→                    # Randomly choose a sample from the minority class
   350→                    random_index = rng.choice(np.where(y[:, i] == 1)[0], size=size)
   351→                except ValueError as e:
   352→                    raise get_empty_class_exception()(index=i) from e
   353→
   354→                # Apply SMOTE
   355→                x_app, y_app = apply(x, y, random_index)
   356→                y_temp.append(y_app)
   357→                x_temp.append(x_app)
   358→
   359→    return x_temp, y_temp

   362→def upsampling(x: np.ndarray, y: np.ndarray, ratio=0.5, mode="repeat"):
   363→    """Balance data through upsampling.
   364→
   365→    We upsample minority classes to have at least 10% (ratio=0.1) of the samples of the majority class.
   366→
   367→    Args:
   368→        x: Samples.
   369→        y: One-hot labels.
   370→        ratio: The minimum ratio of minority to majority samples.
   371→        mode: The upsampling mode. Either 'repeat', 'mean', 'linear' or 'smote'.
   372→
   373→    Returns:
   374→        Upsampled data.
   375→    """
   376→
   377→    # Set numpy random seed
   378→    rng = np.random.default_rng(cfg.RANDOM_SEED)
   379→
   380→    # Determine min number of samples
   381→    min_samples = int(max(y.sum(axis=0), len(y) - y.sum(axis=0)) * ratio) if cfg.BINARY_CLASSIFICATION else int(np.max(y.sum(axis=0)) * ratio)
   382→
   383→    x_temp = []
   384→    y_temp = []
   385→
   386→    if mode == "repeat":
   387→
   388→        def applyRepeat(x, y, random_index):
   389→            return x[random_index[0]], y[random_index[0]]
   390→
   391→        x_temp, y_temp = upsample_core(x, y, min_samples, applyRepeat, size=1)
   392→
   393→    elif mode == "mean":
   394→        # For each class with less than min_samples
   395→        # select two random samples and calculate the mean
   396→        def applyMean(x, y, random_indices):
   397→            # Calculate the mean of the two samples
   398→            mean = np.mean(x[random_indices], axis=0)
   399→
   400→            # Append the mean and label to a temp list
   401→            return mean, y[random_indices[0]]
   402→
   403→        x_temp, y_temp = upsample_core(x, y, min_samples, applyMean)
   404→
   405→    elif mode == "linear":
   406→        # For each class with less than min_samples
   407→        # select two random samples and calculate the linear combination
   408→        def applyLinearCombination(x, y, random_indices):
   409→            # Calculate the linear combination of the two samples
   410→            alpha = rng.uniform(0, 1)
   411→            new_sample = alpha * x[random_indices[0]] + (1 - alpha) * x[random_indices[1]]
   412→
   413→            # Append the new sample and label to a temp list
   414→            return new_sample, y[random_indices[0]]
   415→
   416→        x_temp, y_temp = upsample_core(x, y, min_samples, applyLinearCombination)
   417→
   418→    elif mode == "smote":
   419→        # For each class with less than min_samples apply SMOTE
   420→        def applySmote(x, y, random_index, k=5):
   421→            # Get the k nearest neighbors
   422→            distances = np.sqrt(np.sum((x - x[random_index[0]]) ** 2, axis=1))
   423→            indices = np.argsort(distances)[1 : k + 1]
   424→
   425→            # Randomly choose one of the neighbors
   426→            random_neighbor = rng.choice(indices)
   427→
   428→            # Calculate the difference vector
   429→            diff = x[random_neighbor] - x[random_index[0]]
   430→
   431→            # Randomly choose a weight between 0 and 1
   432→            weight = rng.uniform(0, 1)
   433→
   434→            # Calculate the new sample
   435→            new_sample = x[random_index[0]] + weight * diff
   436→
   437→            # Append the new sample and label to a temp list
   438→            return new_sample, y[random_index[0]]
   439→
   440→        x_temp, y_temp = upsample_core(x, y, min_samples, applySmote, size=1)
   441→
   442→    # Append the temp list to the original data
   443→    if len(x_temp) > 0:
   444→        x = np.vstack((x, np.array(x_temp)))
   445→        y = np.vstack((y, np.array(y_temp)))
   446→
   447→    # Shuffle data
   448→    indices = np.arange(len(x))
   449→    rng.shuffle(indices)
   450→    x = x[indices]
   451→    y = y[indices]
   452→
   453→    del x_temp
   454→    del y_temp
   455→
   456→    return x, y

   517→def load_model(class_output=True):
   518→    """
   519→    Loads the machine learning model based on the configuration provided.
   520→    This function loads either a TensorFlow Lite (TFLite) model or a protobuf model
   521→    depending on the file extension of the model path specified in the configuration.
   522→    It sets up the global variables for the model interpreter and input/output layer indices.
   523→
   524→    Args:
   525→        class_output (bool): If True, sets the output layer index to the classification output.
   526→                             If False, sets the output layer index to the feature embeddings.
   527→    """
   528→    global PBMODEL
   529→    global INTERPRETER
   530→    global INPUT_LAYER_INDEX
   531→    global OUTPUT_LAYER_INDEX
   532→    global OUTPUT_DETAILS
   533→
   534→    if cfg.MODEL_PATH is None:
   535→        raise ValueError("MODEL_PATH is not set.")
   536→
   537→    # Do we have to load the tflite or protobuf model?
   538→    if cfg.MODEL_PATH.endswith(".tflite"):
   539→        if not INTERPRETER:
   540→            # Load TFLite model and allocate tensors.
   541→            INTERPRETER = _load_interpreter(os.path.join(SCRIPT_DIR, cfg.MODEL_PATH), cfg.TFLITE_THREADS)
   542→            INTERPRETER.allocate_tensors()
   543→
   544→            # Get input and output tensors.
   545→            input_details = INTERPRETER.get_input_details()
   546→            OUTPUT_DETAILS = INTERPRETER.get_output_details()
   547→
   548→            # Get input tensor index
   549→            INPUT_LAYER_INDEX = input_details[0]["index"]
   550→
   551→        # Get classification output or feature embeddings
   552→        OUTPUT_LAYER_INDEX = OUTPUT_DETAILS[0]["index"] if class_output else OUTPUT_DETAILS[0]["index"] - 1
   553→
   554→    elif not PBMODEL:
   555→        # Load protobuf model
   556→        # Note: This will throw a bunch of warnings about custom gradients
   557→        # which we will ignore until TF lets us block them
   558→        # TODO: pretty sure this doesn't work anymore, but I don't think anyone uses it
   559→        PBMODEL = keras.models.load_model(os.path.join(SCRIPT_DIR, cfg.MODEL_PATH), compile=False)

   564→    Loads a custom classifier model based on the file extension of the provided model path.
   565→    If the model file ends with ".tflite", it loads a TensorFlow Lite model and sets up the interpreter,
   566→    input layer index, output layer index, and input size.
   567→    If the model file does not end with ".tflite", it loads a TensorFlow SavedModel.
   568→    """
   569→    global C_INTERPRETER
   570→    global C_INPUT_LAYER_INDEX
   571→    global C_OUTPUT_LAYER_INDEX
   572→    global C_INPUT_SIZE
   573→    global C_PBMODEL

   620→def build_linear_classifier(num_labels, input_size, hidden_units=0, dropout=0.0):
   621→    """Builds a classifier.
   622→
   623→    Args:
   624→        num_labels: Output size.
   625→        input_size: Size of the input.
   626→        hidden_units: If > 0, creates another hidden layer with the given number of units.
   627→        dropout: Dropout rate.
   628→
   629→    Returns:
   630→        A new classifier.
   631→    """
   632→    # Build a simple one- or two-layer linear classifier
   633→    model = keras.Sequential()
   634→
   635→    # Input layer
   636→    model.add(keras.layers.InputLayer(input_shape=(input_size,)))
   637→
   638→    # Optional L2 regularization for all dense layers
   639→    regularizer = keras.regularizers.l2(1e-5)
   640→
   641→    # Hidden layer with improved architecture
   642→    if hidden_units > 0:
   643→        # Dropout layer before hidden layer
   644→        if dropout > 0:
   645→            model.add(keras.layers.Dropout(dropout))
   646→
   647→        # Add a hidden layer with L2 regularization
   648→        model.add(keras.layers.Dense(hidden_units, activation="relu", kernel_regularizer=regularizer, kernel_initializer="he_normal"))
   649→
   650→    # Dropout layer before output
   651→    if dropout > 0:
   652→        model.add(keras.layers.Dropout(dropout))
   653→
   654→    # Classification layer with L2 regularization
   655→    model.add(keras.layers.Dense(num_labels, kernel_regularizer=regularizer, kernel_initializer="glorot_uniform"))
   656→
   657→    # Activation layer
   658→    model.add(keras.layers.Activation("sigmoid"))
   659→
   660→    return model

   663→def train_linear_classifier(
   664→    classifier,
   665→    x_train,
   666→    y_train,
   667→    x_test,
   668→    y_test,
   669→    epochs,
   670→    batch_size,
   671→    learning_rate,
   672→    val_split,
   673→    upsampling_ratio,
   674→    upsampling_mode,
   675→    train_with_mixup,
   676→    train_with_label_smoothing,
   677→    train_with_focal_loss=False,
   678→    focal_loss_gamma=2.0,
   679→    focal_loss_alpha=0.25,
   680→    on_epoch_end=None,
   681→):
   682→    """Trains a custom classifier.
   683→
   684→    Trains a new classifier for BirdNET based on the given data.
   685→
   686→    Args:
   687→        classifier: The classifier to be trained.
   688→        x_train: Samples.
   689→        y_train: Labels.
   690→        x_test: Validation samples.
   691→        y_test: Validation labels.
   692→        epochs: Number of epochs to train.
   693→        batch_size: Batch size.
   694→        learning_rate: The learning rate during training.
   695→        val_split: Validation split ratio (is 0 when using test data).
   696→        upsampling_ratio: Upsampling ratio.
   697→        upsampling_mode: Upsampling mode.
   698→        train_with_mixup: If True, applies mixup to the training data.
   699→        train_with_label_smoothing: If True, applies label smoothing to the training data.
   700→        train_with_focal_loss: If True, uses focal loss instead of binary cross-entropy loss.
   701→        focal_loss_gamma: Focal loss gamma parameter.
   702→        focal_loss_alpha: Focal loss alpha parameter.
   703→        on_epoch_end: Optional callback `function(epoch, logs)`.
   704→
   705→    Returns:
   706→        (classifier, history)
   707→    """
   708→
   709→    class FunctionCallback(keras.callbacks.Callback):
   710→        def __init__(self, on_epoch_end=None) -> None:
   711→            super().__init__()
   712→            self.on_epoch_end_fn = on_epoch_end
   713→
   714→        def on_epoch_end(self, epoch, logs=None):
   715→            if self.on_epoch_end_fn:
   716→                self.on_epoch_end_fn(epoch, logs)
   717→
   718→    # Set random seed
   719→    rng = np.random.default_rng(cfg.RANDOM_SEED)
   720→
   721→    # Shuffle data
   722→    idx = np.arange(x_train.shape[0])
   723→    rng.shuffle(idx)
   724→    x_train = x_train[idx]
   725→    y_train = y_train[idx]
   726→
   727→    # Random val split
   728→    if val_split > 0:
   729→        if not cfg.MULTI_LABEL:
   730→            x_train, y_train, x_val, y_val = random_split(x_train, y_train, val_split)
   731→        else:
   732→            x_train, y_train, x_val, y_val = random_multilabel_split(x_train, y_train, val_split)
   733→    else:
   734→        x_val = x_test
   735→        y_val = y_test
   736→
   737→    print(
   738→        f"Training on {x_train.shape[0]} samples, validating on {x_val.shape[0]} samples.",
   739→        flush=True,
   740→    )
   741→
   742→    # Upsample training data
   743→    if upsampling_ratio > 0:
   744→        x_train, y_train = upsampling(x_train, y_train, upsampling_ratio, upsampling_mode)
   745→        print(f"Upsampled training data to {x_train.shape[0]} samples.", flush=True)
   746→
   747→    # Apply mixup to training data
   748→    if train_with_mixup and not cfg.BINARY_CLASSIFICATION:
   749→        x_train, y_train = mixup(x_train, y_train)
   750→
   751→    # Apply label smoothing
   752→    if train_with_label_smoothing and not cfg.BINARY_CLASSIFICATION:
   753→        y_train = label_smoothing(y_train)
   754→
   755→    # Early stopping with patience depending on dataset size
   756→    patience = min(10, max(5, int(epochs / 10)))
   757→    min_delta = 0.001
   758→
   759→    callbacks = [
   760→        # EarlyStopping with restore_best_weights
   761→        keras.callbacks.EarlyStopping(
   762→            monitor="val_loss",
   763→            mode="min",
   764→            patience=patience,
   765→            verbose=1,
   766→            min_delta=min_delta,
   767→            restore_best_weights=True,
   768→        ),
   769→        # Function callback for progress tracking
   770→        FunctionCallback(on_epoch_end=on_epoch_end),
   771→    ]
   772→
   773→    # Learning rate schedule - use cosine decay with warmup
   774→    warmup_epochs = min(5, int(epochs * 0.1))
   775→
   776→    def lr_schedule(epoch, lr):
   777→        if epoch < warmup_epochs:
   778→            # Linear warmup
   779→            return learning_rate * (epoch + 1) / warmup_epochs
   780→
   781→        # Cosine decay
   782→        progress = (epoch - warmup_epochs) / (epochs - warmup_epochs)
   783→        return learning_rate * (0.1 + 0.9 * (1 + np.cos(np.pi * progress)) / 2)
   784→
   785→    # Add LR scheduler callback
   786→    callbacks.append(keras.callbacks.LearningRateScheduler(lr_schedule))
   787→
   788→    def _focal_loss(y_true, y_pred):
   789→        return focal_loss(y_true, y_pred, gamma=focal_loss_gamma, alpha=focal_loss_alpha)
   790→
   791→    # Choose the loss function based on config
   792→    loss_function = _focal_loss if train_with_focal_loss else custom_loss
   793→
   794→    # Compile model with appropriate metrics for classification task
   795→    classifier.compile(
   796→        optimizer=keras.optimizers.Adam(learning_rate=learning_rate),
   797→        loss=loss_function,
   798→        metrics=[
   799→            keras.metrics.AUC(
   800→                curve="PR",
   801→                multi_label=cfg.MULTI_LABEL,
   802→                name="AUPRC",
   803→                num_labels=y_train.shape[1] if cfg.MULTI_LABEL else None,
   804→                from_logits=True,
   805→            ),
   806→            keras.metrics.AUC(
   807→                curve="ROC",
   808→                multi_label=cfg.MULTI_LABEL,
   809→                name="AUROC",
   810→                num_labels=y_train.shape[1] if cfg.MULTI_LABEL else None,
   811→                from_logits=True,
   812→            ),
   813→        ],
   814→    )
   815→
   816→    # Train model
   817→    history = classifier.fit(x_train, y_train, epochs=epochs, batch_size=batch_size, validation_data=(x_val, y_val), callbacks=callbacks)
   818→
   819→    return classifier, history

  1021→def focal_loss(y_true, y_pred, gamma=2.0, alpha=0.25, epsilon=1e-7):
  1022→    """
  1023→    Focal loss for better handling of class imbalance.
  1024→
  1025→    This loss function gives more weight to hard examples and down-weights easy examples.
  1026→    Particularly helpful for imbalanced datasets where some classes have few samples.
  1027→
  1028→    Args:
  1029→        y_true: Ground truth labels.
  1030→        y_pred: Predicted probabilities.
  1031→        gamma: Focusing parameter. Higher values mean more focus on hard examples.
  1032→        alpha: Balance parameter. Controls weight of positive vs negative examples.
  1033→        epsilon: Small constant to prevent log(0).
  1034→
  1035→    Returns:
  1036→        Focal loss value.
  1037→    """
  1038→    # Apply sigmoid if not already applied
  1039→    y_pred = tf.clip_by_value(y_pred, epsilon, 1.0 - epsilon)
  1040→
  1041→    # Calculate cross entropy
  1042→    cross_entropy = -y_true * tf.math.log(y_pred) - (1 - y_true) * tf.math.log(1 - y_pred)
  1043→
  1044→    # Calculate focal weight
  1045→    p_t = y_true * y_pred + (1 - y_true) * (1 - y_pred)
  1046→    focal_weight = tf.pow(1 - p_t, gamma)
  1047→
  1048→    # Apply alpha balancing
  1049→    alpha_factor = y_true * alpha + (1 - y_true) * (1 - alpha)
  1050→
  1051→    # Calculate focal loss
  1052→    focal_loss = alpha_factor * focal_weight * cross_entropy
  1053→
  1054→    # Sum over all classes
  1055→    return tf.reduce_sum(focal_loss, axis=-1)
  1056→
  1057→
  1058→def custom_loss(y_true, y_pred, epsilon=1e-7):
  1059→    # Calculate loss for positive labels with epsilon
  1060→    positive_loss = -tf.reduce_sum(y_true * tf.math.log(tf.clip_by_value(y_pred, epsilon, 1.0 - epsilon)), axis=-1)
  1061→
  1062→    # Calculate loss for negative labels with epsilon
  1063→    negative_loss = -tf.reduce_sum((1 - y_true) * tf.math.log(tf.clip_by_value(1 - y_pred, epsilon, 1.0 - epsilon)), axis=-1)
  1064→
  1065→    # Combine both loss terms
  1066→    return positive_loss + negative_loss

  1069→def flat_sigmoid(x, sensitivity=-1, bias=1.0):
  1070→    """
  1071→    Applies a flat sigmoid function to the input array with a bias shift.
  1072→
  1073→    The flat sigmoid function is defined as:
  1074→        f(x) = 1 / (1 + exp(sensitivity * clip(x + bias, -20, 20)))
  1075→
  1076→    We transform the bias parameter to a range of [-100, 100] with the formula:
  1077→        transformed_bias = (bias - 1.0) * 10.0
  1078→
  1079→    Thus, higher bias values will shift the sigmoid function to the right on the x-axis, making it more "sensitive".
  1080→
  1081→    Note: Not sure why we are clipping, must be for numerical stability somewhere else in the code.
  1082→
  1083→    Args:
  1084→        x (array-like): Input data.
  1085→        sensitivity (float, optional): Sensitivity parameter for the sigmoid function. Default is -1.
  1086→        bias (float, optional): Bias parameter to shift the sigmoid function on the x-axis. Must be in the range [0.01, 1.99]. Default is 1.0.
  1087→
  1088→    Returns:
  1089→        numpy.ndarray: Transformed data after applying the flat sigmoid function.
  1090→    """
  1091→
  1092→    transformed_bias = (bias - 1.0) * 10.0
  1093→
  1094→    return 1 / (1.0 + np.exp(sensitivity * np.clip(x + transformed_bias, -20, 20)))
