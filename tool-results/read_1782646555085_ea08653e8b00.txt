    84‚Üí    84‚Üídef label_smoothing(y: np.ndarray, alpha=0.1):
    85‚Üí    85‚Üí    """
    86‚Üí    86‚Üí    Applies label smoothing to the given labels.
    87‚Üí    87‚Üí    Label smoothing is a technique used to prevent the model from becoming overconfident by adjusting the target labels.
    88‚Üí    88‚Üí    It subtracts a small value (alpha) from the correct label and distributes it among the other labels.
    89‚Üí    89‚Üí    Args:
    90‚Üí    90‚Üí        y (numpy.ndarray): Array of labels to be smoothed. The array should be of shape (num_labels,).
    91‚Üí    91‚Üí        alpha (float, optional): Smoothing parameter. Default is 0.1.
    92‚Üí    92‚Üí    Returns:
    93‚Üí    93‚Üí        numpy.ndarray: The smoothed labels.
    94‚Üí    94‚Üí    """
    95‚Üí    95‚Üí    # Subtract alpha from correct label when it is >0
    96‚Üí    96‚Üí    y[y > 0] -= alpha
    97‚Üí    97‚Üí
    98‚Üí    98‚Üí    # Assigned alpha to all other labels
    99‚Üí    99‚Üí    y[y == 0] = alpha / y.shape[0]
   100‚Üí   100‚Üí
   101‚Üí   101‚Üí    return y

   104‚Üí   104‚Üídef mixup(x, y, augmentation_ratio=0.25, alpha=0.2):
   105‚Üí   105‚Üí    """Apply mixup to the given data.
   106‚Üí   106‚Üí
   107‚Üí   107‚Üí    Mixup is a data augmentation technique that generates new samples by
   108‚Üí   108‚Üí    mixing two samples and their labels.
   109‚Üí   109‚Üí
   110‚Üí   110‚Üí    Args:
   111‚Üí   111‚Üí        x: Samples.
   112‚Üí   112‚Üí        y: One-hot labels.
   113‚Üí   113‚Üí        augmentation_ratio: The ratio of augmented samples.
   114‚Üí   114‚Üí        alpha: The beta distribution parameter.
   115‚Üí   115‚Üí
   116‚Üí   116‚Üí    Returns:
   117‚Üí   117‚Üí        Augmented data.
   118‚Üí   118‚Üí    """
   119‚Üí   119‚Üí    rng = np.random.default_rng(cfg.RANDOM_SEED)
   120‚Üí   120‚Üí
   121‚Üí   121‚Üí    # Get indices of all positive samples
   122‚Üí   122‚Üí    positive_indices = np.unique(np.where(y[:, :] == 1)[0])
   123‚Üí   123‚Üí
   124‚Üí   124‚Üí    # Calculate the number of samples to augment based on the ratio
   125‚Üí   125‚Üí    num_samples_to_augment = int(len(positive_indices) * augmentation_ratio)
   126‚Üí   126‚Üí
   127‚Üí   127‚Üí    # Indices of samples, that are already mixed up
   128‚Üí   128‚Üí    mixed_up_indices = []
   129‚Üí   129‚Üí
   130‚Üí   130‚Üí    for _ in range(num_samples_to_augment):
   131‚Üí   131‚Üí        # Randomly choose one instance from the positive samples
   132‚Üí   132‚Üí        index = rng.choice(positive_indices)
   133‚Üí   133‚Üí
   134‚Üí   134‚Üí        # Choose another one, when the chosen one was already mixed up
   135‚Üí   135‚Üí        while index in mixed_up_indices:
   136‚Üí   136‚Üí            index = rng.choice(positive_indices)
   137‚Üí   137‚Üí
   138‚Üí   138‚Üí        x1, y1 = x[index], y[index]
   139‚Üí   139‚Üí
   140‚Üí   140‚Üí        # Randomly choose a different instance from the dataset
   141‚Üí   141‚Üí        second_index = rng.choice(positive_indices)
   142‚Üí   142‚Üí
   143‚Üí   143‚Üí        # Choose again, when the same or an already mixed up sample was selected
   144‚Üí   144‚Üí        while second_index == index or second_index in mixed_up_indices:
   145‚Üí   145‚Üí            second_index = rng.choice(positive_indices)
   146‚Üí   146‚Üí        x2, y2 = x[second_index], y[second_index]
   147‚Üí   147‚Üí
   148‚Üí   148‚Üí        # Generate a random mixing coefficient (lambda)
   149‚Üí   149‚Üí        lambda_ = rng.beta(alpha, alpha)
   150‚Üí   150‚Üí
   151‚Üí   151‚Üí        # Mix the embeddings and labels
   152‚Üí   152‚Üí        mixed_x = lambda_ * x1 + (1 - lambda_) * x2
   153‚Üí   153‚Üí        mixed_y = lambda_ * y1 + (1 - lambda_) * y2
   154‚Üí   154‚Üí
   155‚Üí   155‚Üí        # Replace one of the original samples and labels with the augmented sample and labels
   156‚Üí   156‚Üí        x[index] = mixed_x
   157‚Üí   157‚Üí        y[index] = mixed_y
   158‚Üí   158‚Üí
   159‚Üí   159‚Üí        # Mark the sample as already mixed up
   160‚Üí   160‚Üí        mixed_up_indices.append(index)
   161‚Üí   161‚Üí
   162‚Üí   162‚Üí        del mixed_x
   163‚Üí   163‚Üí        del mixed_y
   164‚Üí   164‚Üí
   165‚Üí   165‚Üí    return x, y

   168‚Üí   168‚Üídef random_split(x, y, val_ratio=0.2):
   169‚Üí   169‚Üí    """Splits the data into training and validation data.
   170‚Üí   170‚Üí
   171‚Üí   171‚Üí    Makes sure that each class is represented in both sets.
   172‚Üí   172‚Üí
   173‚Üí   173‚Üí    Args:
   174‚Üí   174‚Üí        x: Samples.
   175‚Üí   175‚Üí        y: One-hot labels.
   176‚Üí   176‚Üí        val_ratio: The ratio of validation data.
   177‚Üí   177‚Üí
   178‚Üí   178‚Üí    Returns:
   179‚Üí   179‚Üí        A tuple of (x_train, y_train, x_val, y_val).
   180‚Üí   180‚Üí    """
   181‚Üí   181‚Üí    rng = np.random.default_rng(cfg.RANDOM_SEED)
   182‚Üí   182‚Üí
   183‚Üí   183‚Üí    # Get number of classes
   184‚Üí   184‚Üí    num_classes = y.shape[1]
   185‚Üí   185‚Üí
   186‚Üí   186‚Üí    # Initialize training and validation data
   187‚Üí   187‚Üí    x_train, y_train, x_val, y_val = [], [], [], []
   188‚Üí   188‚Üí
   189‚Üí   189‚Üí    # Split data
   190‚Üí   190‚Üí    for i in range(num_classes):
   191‚Üí   191‚Üí        # Get indices of positive samples of current class
   192‚Üí   192‚Üí        positive_indices = np.where(y[:, i] == 1)[0]
   193‚Üí   193‚Üí
   194‚Üí   194‚Üí        # Get indices of negative samples of current class
   195‚Üí   195‚Üí        negative_indices = np.where(y[:, i] == -1)[0]
   196‚Üí   196‚Üí
   197‚Üí   197‚Üí        # Get number of samples for each set
   198‚Üí   198‚Üí        num_samples = len(positive_indices)
   199‚Üí   199‚Üí        num_samples_train = max(1, int(num_samples * (1 - val_ratio)))
   200‚Üí   200‚Üí        num_samples_val = max(0, num_samples - num_samples_train)
   201‚Üí   201‚Üí
   202‚Üí   202‚Üí        # Randomly choose samples for training and validation
   203‚Üí   203‚Üí        rng.shuffle(positive_indices)
   204‚Üí   204‚Üí        train_indices = positive_indices[:num_samples_train]
   205‚Üí   205‚Üí        val_indices = positive_indices[num_samples_train : num_samples_train + num_samples_val]
   206‚Üí   206‚Üí
   207‚Üí   207‚Üí        # Append samples to training and validation data
   208‚Üí   208‚Üí        x_train.append(x[train_indices])
   209‚Üí   209‚Üí        y_train.append(y[train_indices])
   210‚Üí   210‚Üí        x_val.append(x[val_indices])
   211‚Üí   211‚Üí        y_val.append(y[val_indices])
   212‚Üí   212‚Üí
   213‚Üí   213‚Üí        # Append negative samples to training data
   214‚Üí   214‚Üí        x_train.append(x[negative_indices])
   215‚Üí   215‚Üí        y_train.append(y[negative_indices])
   216‚Üí   216‚Üí
   217‚Üí   217‚Üí    # Add samples for non-event classes to training and validation data
   218‚Üí   218‚Üí    non_event_indices = np.where(np.sum(y[:, :], axis=1) == 0)[0]
   219‚Üí   219‚Üí    num_samples = len(non_event_indices)
   220‚Üí   220‚Üí    num_samples_train = max(1, int(num_samples * (1 - val_ratio)))
   221‚Üí   221‚Üí    num_samples_val = max(0, num_samples - num_samples_train)
   222‚Üí   222‚Üí    rng.shuffle(non_event_indices)
   223‚Üí   223‚Üí    train_indices = non_event_indices[:num_samples_train]
   224‚Üí   224‚Üí    val_indices = non_event_indices[num_samples_train : num_samples_train + num_samples_val]
   225‚Üí   225‚Üí    x_train.append(x[train_indices])
   226‚Üí   226‚Üí    y_train.append(y[train_indices])
   227‚Üí   227‚Üí    x_val.append(x[val_indices])
   228‚Üí   228‚Üí    y_val.append(y[val_indices])
   229‚Üí   229‚Üí
   230‚Üí   230‚Üí    # Concatenate data
   231‚Üí   231‚Üí    x_train = np.concatenate(x_train)
   232‚Üí   232‚Üí    y_train = np.concatenate(y_train)
   233‚Üí   233‚Üí    x_val = np.concatenate(x_val)
   234‚Üí   234‚Üí    y_val = np.concatenate(y_val)
   235‚Üí   235‚Üí
   236‚Üí   236‚Üí    # Shuffle data
   237‚Üí   237‚Üí    indices = np.arange(len(x_train))
   238‚Üí   238‚Üí    rng.shuffle(indices)
   239‚Üí   239‚Üí    x_train = x_train[indices]
   240‚Üí   240‚Üí    y_train = y_train[indices]
   241‚Üí   241‚Üí
   242‚Üí   242‚Üí    indices = np.arange(len(x_val))
   243‚Üí   243‚Üí    rng.shuffle(indices)
   244‚Üí   244‚Üí    x_val = x_val[indices]
   245‚Üí   245‚Üí    y_val = y_val[indices]
   246‚Üí   246‚Üí
   247‚Üí   247‚Üí    return x_train, y_train, x_val, y_val

   250‚Üí   250‚Üídef random_multilabel_split(x, y, val_ratio=0.2):
   251‚Üí   251‚Üí    """Splits the data into training and validation data.
   252‚Üí   252‚Üí
   253‚Üí   253‚Üí    Makes sure that each combination of classes is represented in both sets.
   254‚Üí   254‚Üí
   255‚Üí   255‚Üí    Args:
   256‚Üí   256‚Üí        x: Samples.
   257‚Üí   257‚Üí        y: One-hot labels.
   258‚Üí   258‚Üí        val_ratio: The ratio of validation data.
   259‚Üí   259‚Üí
   260‚Üí   260‚Üí    Returns:
   261‚Üí   261‚Üí        A tuple of (x_train, y_train, x_val, y_val).
   262‚Üí   262‚Üí
   263‚Üí   263‚Üí    """
   264‚Üí   264‚Üí    rng = np.random.default_rng(cfg.RANDOM_SEED)
   265‚Üí   265‚Üí
   266‚Üí   266‚Üí    # Find all combinations of labels
   267‚Üí   267‚Üí    class_combinations = np.unique(y, axis=0)
   268‚Üí   268‚Üí
   269‚Üí   269‚Üí    # Initialize training and validation data
   270‚Üí   270‚Üí    x_train, y_train, x_val, y_val = [], [], [], []
   271‚Üí   271‚Üí
   272‚Üí   272‚Üí    # Split the data for each combination of labels
   273‚Üí   273‚Üí    for class_combination in class_combinations:
   274‚Üí   274‚Üí        # find all indices
   275‚Üí   275‚Üí        indices = np.where((y == class_combination).all(axis=1))[0]
   276‚Üí   276‚Üí
   277‚Üí   277‚Üí        # When negative sample use only for training
   278‚Üí   278‚Üí        if -1 in class_combination:
   279‚Üí   279‚Üí            x_train.append(x[indices])
   280‚Üí   280‚Üí            y_train.append(y[indices])
   281‚Üí   281‚Üí        # Otherwise split according to the validation split
   282‚Üí   282‚Üí        else:
   283‚Üí   283‚Üí            # Get number of samples for each set
   284‚Üí   284‚Üí            num_samples = len(indices)
   285‚Üí   285‚Üí            num_samples_train = max(1, int(num_samples * (1 - val_ratio)))
   286‚Üí   286‚Üí            num_samples_val = max(0, num_samples - num_samples_train)
   287‚Üí   287‚Üí            # Randomly choose samples for training and validation
   288‚Üí   288‚Üí            rng.shuffle(indices)
   289‚Üí   289‚Üí            train_indices = indices[:num_samples_train]
   290‚Üí   290‚Üí            val_indices = indices[num_samples_train : num_samples_train + num_samples_val]
   291‚Üí   291‚Üí            # Append samples to training and validation data
   292‚Üí   292‚Üí            x_train.append(x[train_indices])
   293‚Üí   293‚Üí            y_train.append(y[train_indices])
   294‚Üí   294‚Üí            x_val.append(x[val_indices])
   295‚Üí   295‚Üí            y_val.append(y[val_indices])
   296‚Üí   296‚Üí
   297‚Üí   297‚Üí    # Concatenate data
   298‚Üí   298‚Üí    x_train = np.concatenate(x_train)
   299‚Üí   299‚Üí    y_train = np.concatenate(y_train)
   300‚Üí   300‚Üí    x_val = np.concatenate(x_val)
   301‚Üí   301‚Üí    y_val = np.concatenate(y_val)
   302‚Üí   302‚Üí
   303‚Üí   303‚Üí    # Shuffle data
   304‚Üí   304‚Üí    indices = np.arange(len(x_train))
   305‚Üí   305‚Üí    rng.shuffle(indices)
   306‚Üí   306‚Üí    x_train = x_train[indices]
   307‚Üí   307‚Üí    y_train = y_train[indices]
   308‚Üí   308‚Üí
   309‚Üí   309‚Üí    indices = np.arange(len(x_val))
   310‚Üí   310‚Üí    rng.shuffle(indices)
   311‚Üí   311‚Üí    x_val = x_val[indices]
   312‚Üí   312‚Üí    y_val = y_val[indices]
   313‚Üí   313‚Üí
   314‚Üí   314‚Üí    return x_train, y_train, x_val, y_val

   317‚Üí   317‚Üídef upsample_core(x: np.ndarray, y: np.ndarray, min_samples: int, apply, size=2):
   318‚Üí   318‚Üí    """
   319‚Üí   319‚Üí    Upsamples the minority class in the dataset using the specified apply function.
   320‚Üí   320‚Üí    Parameters:
   321‚Üí   321‚Üí        x (np.ndarray): The feature matrix.
   322‚Üí   322‚Üí        y (np.ndarray): The target labels.
   323‚Üí   323‚Üí        min_samples (int): The minimum number of samples required for the minority class.
   324‚Üí   324‚Üí        apply (callable): A function that applies the SMOTE or any other algorithm to the data.
   325‚Üí   325‚Üí        size (int, optional): The number of samples to generate in each iteration. Default is 2.
   326‚Üí   326‚Üí    Returns:
   327‚Üí   327‚Üí        tuple: A tuple containing the upsampled feature matrix and target labels.
   328‚Üí   328‚Üí    """
   329‚Üí   329‚Üí    rng = np.random.default_rng(cfg.RANDOM_SEED)
   330‚Üí   330‚Üí    y_temp = []
   331‚Üí   331‚Üí    x_temp = []
   332‚Üí   332‚Üí
   333‚Üí   333‚Üí    if cfg.BINARY_CLASSIFICATION:
   334‚Üí   334‚Üí        # Determine if 1 or 0 is the minority class
   335‚Üí   335‚Üí        minority_label = 1 if y.sum(axis=0) < len(y) - y.sum(axis=0) else 0
   336‚Üí   336‚Üí
   337‚Üí   337‚Üí        while np.where(y == minority_label)[0].shape[0] + len(y_temp) < min_samples:
   338‚Üí   338‚Üí            # Randomly choose a sample from the minority class
   339‚Üí   339‚Üí            random_index = rng.choice(np.where(y == minority_label)[0], size=size)
   340‚Üí   340‚Üí
   341‚Üí   341‚Üí            # Apply SMOTE
   342‚Üí   342‚Üí            x_app, y_app = apply(x, y, random_index)
   343‚Üí   343‚Üí            y_temp.append(y_app)
   344‚Üí   344‚Üí            x_temp.append(x_app)
   345‚Üí   345‚Üí    else:
   346‚Üí   346‚Üí        for i in range(y.shape[1]):
   347‚Üí   347‚Üí            while y[:, i].sum() + len(y_temp) < min_samples:
   348‚Üí   348‚Üí                try:
   349‚Üí   349‚Üí                    # Randomly choose a sample from the minority class
   350‚Üí   350‚Üí                    random_index = rng.choice(np.where(y[:, i] == 1)[0], size=size)
   351‚Üí   351‚Üí                except ValueError as e:
   352‚Üí   352‚Üí                    raise get_empty_class_exception()(index=i) from e
   353‚Üí   353‚Üí
   354‚Üí   354‚Üí                # Apply SMOTE
   355‚Üí   355‚Üí                x_app, y_app = apply(x, y, random_index)
   356‚Üí   356‚Üí                y_temp.append(y_app)
   357‚Üí   357‚Üí                x_temp.append(x_app)
   358‚Üí   358‚Üí
   359‚Üí   359‚Üí    return x_temp, y_temp
   360‚Üí   360‚Üí
   361‚Üí   361‚Üí
   362‚Üí   362‚Üídef upsampling(x: np.ndarray, y: np.ndarray, ratio=0.5, mode="repeat"):
   363‚Üí   363‚Üí    """Balance data through upsampling.
   364‚Üí   364‚Üí
   365‚Üí   365‚Üí    We upsample minority classes to have at least 10% (ratio=0.1) of the samples of the majority class.
   366‚Üí   366‚Üí
   367‚Üí   367‚Üí    Args:
   368‚Üí   368‚Üí        x: Samples.
   369‚Üí   369‚Üí        y: One-hot labels.
   370‚Üí   370‚Üí        ratio: The minimum ratio of minority to majority samples.
   371‚Üí   371‚Üí        mode: The upsampling mode. Either 'repeat', 'mean', 'linear' or 'smote'.
   372‚Üí   372‚Üí
   373‚Üí   373‚Üí    Returns:
   374‚Üí   374‚Üí        Upsampled data.
   375‚Üí   375‚Üí    """
   376‚Üí   376‚Üí
   377‚Üí   377‚Üí    # Set numpy random seed
   378‚Üí   378‚Üí    rng = np.random.default_rng(cfg.RANDOM_SEED)
   379‚Üí   379‚Üí
   380‚Üí   380‚Üí    # Determine min number of samples
   381‚Üí   381‚Üí    min_samples = int(max(y.sum(axis=0), len(y) - y.sum(axis=0)) * ratio) if cfg.BINARY_CLASSIFICATION else int(np.max(y.sum(axis=0)) * ratio)
   382‚Üí   382‚Üí
   383‚Üí   383‚Üí    x_temp = []
   384‚Üí   384‚Üí    y_temp = []
   385‚Üí   385‚Üí
   386‚Üí   386‚Üí    if mode == "repeat":
   387‚Üí   387‚Üí
   388‚Üí   388‚Üí        def applyRepeat(x, y, random_index):
   389‚Üí   389‚Üí            return x[random_index[0]], y[random_index[0]]
   390‚Üí   390‚Üí
   391‚Üí   391‚Üí        x_temp, y_temp = upsample_core(x, y, min_samples, applyRepeat, size=1)
   392‚Üí   392‚Üí
   393‚Üí   393‚Üí    elif mode == "mean":
   394‚Üí   394‚Üí        # For each class with less than min_samples
   395‚Üí   395‚Üí        # select two random samples and calculate the mean
   396‚Üí   396‚Üí        def applyMean(x, y, random_indices):
   397‚Üí   397‚Üí            # Calculate the mean of the two samples
   398‚Üí   398‚Üí            mean = np.mean(x[random_indices], axis=0)
   399‚Üí   399‚Üí
   400‚Üí   400‚Üí            # Append the mean and label to a temp list
   401‚Üí   401‚Üí            return mean, y[random_indices[0]]
   402‚Üí   402‚Üí
   403‚Üí   403‚Üí        x_temp, y_temp = upsample_core(x, y, min_samples, applyMean)
   404‚Üí   404‚Üí
   405‚Üí   405‚Üí    elif mode == "linear":
   406‚Üí   406‚Üí        # For each class with less than min_samples
   407‚Üí   407‚Üí        # select two random samples and calculate the linear combination
   408‚Üí   408‚Üí        def applyLinearCombination(x, y, random_indices):
   409‚Üí   409‚Üí            # Calculate the linear combination of the two samples
   410‚Üí   410‚Üí            alpha = rng.uniform(0, 1)
   411‚Üí   411‚Üí            new_sample = alpha * x[random_indices[0]] + (1 - alpha) * x[random_indices[1]]
   412‚Üí   412‚Üí
   413‚Üí   413‚Üí            # Append the new sample and label to a temp list
   414‚Üí   414‚Üí            return new_sample, y[random_indices[0]]
   415‚Üí   415‚Üí
   416‚Üí   416‚Üí        x_temp, y_temp = upsample_core(x, y, min_samples, applyLinearCombination)
   417‚Üí   417‚Üí
   418‚Üí   418‚Üí    elif mode == "smote":
   419‚Üí   419‚Üí        # For each class with less than min_samples apply SMOTE
   420‚Üí   420‚Üí        def applySmote(x, y, random_index, k=5):
   421‚Üí   421‚Üí            # Get the k nearest neighbors
   422‚Üí   422‚Üí            distances = np.sqrt(np.sum((x - x[random_index[0]]) ** 2, axis=1))
   423‚Üí   423‚Üí            indices = np.argsort(distances)[1 : k + 1]
   424‚Üí   424‚Üí
   425‚Üí   425‚Üí            # Randomly choose one of the neighbors
   426‚Üí   426‚Üí            random_neighbor = rng.choice(indices)
   427‚Üí   427‚Üí
   428‚Üí   428‚Üí            # Calculate the difference vector
   429‚Üí   429‚Üí            diff = x[random_neighbor] - x[random_index[0]]
   430‚Üí   430‚Üí
   431‚Üí   431‚Üí            # Randomly choose a weight between 0 and 1
   432‚Üí   432‚Üí            weight = rng.uniform(0, 1)
   433‚Üí   433‚Üí
   434‚Üí   434‚Üí            # Calculate the new sample
   435‚Üí   435‚Üí            new_sample = x[random_index[0]] + weight * diff
   436‚Üí   436‚Üí
   437‚Üí   437‚Üí            # Append the new sample and label to a temp list
   438‚Üí   438‚Üí            return new_sample, y[random_index[0]]
   439‚Üí   439‚Üí
   440‚Üí   440‚Üí        x_temp, y_temp = upsample_core(x, y, min_samples, applySmote, size=1)
   441‚Üí   441‚Üí
   442‚Üí   442‚Üí    # Append the temp list to the original data
   443‚Üí   443‚Üí    if len(x_temp) > 0:
   444‚Üí   444‚Üí        x = np.vstack((x, np.array(x_temp)))
   445‚Üí   445‚Üí        y = np.vstack((y, np.array(y_temp)))
   446‚Üí   446‚Üí
   447‚Üí   447‚Üí    # Shuffle data
   448‚Üí   448‚Üí    indices = np.arange(len(x))
   449‚Üí   449‚Üí    rng.shuffle(indices)
   450‚Üí   450‚Üí    x = x[indices]
   451‚Üí   451‚Üí    y = y[indices]
   452‚Üí   452‚Üí
   453‚Üí   453‚Üí    del x_temp
   454‚Üí   454‚Üí    del y_temp
   455‚Üí   455‚Üí
   456‚Üí   456‚Üí    return x, y

   620‚Üí   620‚Üídef build_linear_classifier(num_labels, input_size, hidden_units=0, dropout=0.0):
   621‚Üí   621‚Üí    """Builds a classifier.
   622‚Üí   622‚Üí
   623‚Üí   623‚Üí    Args:
   624‚Üí   624‚Üí        num_labels: Output size.
   625‚Üí   625‚Üí        input_size: Size of the input.
   626‚Üí   626‚Üí        hidden_units: If > 0, creates another hidden layer with the given number of units.
   627‚Üí   627‚Üí        dropout: Dropout rate.
   628‚Üí   628‚Üí
   629‚Üí   629‚Üí    Returns:
   630‚Üí   630‚Üí        A new classifier.
   631‚Üí   631‚Üí    """
   632‚Üí   632‚Üí    # Build a simple one- or two-layer linear classifier
   633‚Üí   633‚Üí    model = keras.Sequential()
   634‚Üí   634‚Üí
   635‚Üí   635‚Üí    # Input layer
   636‚Üí   636‚Üí    model.add(keras.layers.InputLayer(input_shape=(input_size,)))
   637‚Üí   637‚Üí
   638‚Üí   638‚Üí    # Optional L2 regularization for all dense layers
   639‚Üí   639‚Üí    regularizer = keras.regularizers.l2(1e-5)
   640‚Üí   640‚Üí
   641‚Üí   641‚Üí    # Hidden layer with improved architecture
   642‚Üí   642‚Üí    if hidden_units > 0:
   643‚Üí   643‚Üí        # Dropout layer before hidden layer
   644‚Üí   644‚Üí        if dropout > 0:
   645‚Üí   645‚Üí            model.add(keras.layers.Dropout(dropout))
   646‚Üí   646‚Üí

   675‚Üí   675‚Üí    train_with_mixup,

   678‚Üí   678‚Üí    focal_loss_gamma=2.0,

   680‚Üí   680‚Üí    on_epoch_end=None,
   681‚Üí   681‚Üí):
   682‚Üí   682‚Üí    """Trains a custom classifier.
   683‚Üí   683‚Üí
   684‚Üí   684‚Üí    Trains a new classifier for BirdNET based on the given data.
   685‚Üí   685‚Üí
   686‚Üí   686‚Üí    Args:
   687‚Üí   687‚Üí        classifier: The classifier to be trained.
   688‚Üí   688‚Üí        x_train: Samples.
   689‚Üí   689‚Üí        y_train: Labels.
   690‚Üí   690‚Üí        x_test: Validation samples.

   692‚Üí   692‚Üí        epochs: Number of epochs to train.
   693‚Üí   693‚Üí        batch_size: Batch size.
   694‚Üí   694‚Üí        learning_rate: The learning rate during training.
   695‚Üí   695‚Üí        val_split: Validation split ratio (is 0 when using test data).
   696‚Üí   696‚Üí        upsampling_ratio: Upsampling ratio.
   697‚Üí   697‚Üí        upsampling_mode: Upsampling mode.
   698‚Üí   698‚Üí        train_with_mixup: If True, applies mixup to the training data.
   699‚Üí   699‚Üí        train_with_label_smoothing: If True, applies label smoothing to the training data.
   700‚Üí   700‚Üí        train_with_focal_loss: If True, uses focal loss instead of binary cross-entropy loss.
   701‚Üí   701‚Üí        focal_loss_gamma: Focal loss gamma parameter.
   702‚Üí   702‚Üí        focal_loss_alpha: Focal loss alpha parameter.
   703‚Üí   703‚Üí        on_epoch_end: Optional callback `function(epoch, logs)`.
   704‚Üí   704‚Üí

   707‚Üí   707‚Üí    """
   708‚Üí   708‚Üí
   709‚Üí   709‚Üí    class FunctionCallback(keras.callbacks.Callback):
   710‚Üí   710‚Üí        def __init__(self, on_epoch_end=None) -> None:
   711‚Üí   711‚Üí            super().__init__()
   712‚Üí   712‚Üí            self.on_epoch_end_fn = on_epoch_end

   717‚Üí   717‚Üí

   720‚Üí   720‚Üí

   726‚Üí   726‚Üí
   727‚Üí   727‚Üí    # Random val split
   728‚Üí   728‚Üí    if val_split > 0:
   729‚Üí   729‚Üí        if not cfg.MULTI_LABEL:
   730‚Üí   730‚Üí            x_train, y_train, x_val, y_val = random_split(x_train, y_train, val_split)
   731‚Üí   731‚Üí        else:
   732‚Üí   732‚Üí            x_train, y_train, x_val, y_val = random_multilabel_split(x_train, y_train, val_split)
   733‚Üí   733‚Üí    else:
   734‚Üí   734‚Üí        x_val = x_test
   735‚Üí   735‚Üí        y_val = y_test
   736‚Üí   736‚Üí
   737‚Üí   737‚Üí    print(
   738‚Üí   738‚Üí        f"Training on {x_train.shape[0]} samples, validating on {x_val.shape[0]} samples.",
   739‚Üí   739‚Üí        flush=True,
   740‚Üí   740‚Üí    )
   741‚Üí   741‚Üí
   742‚Üí   742‚Üí    # Upsample training data
   743‚Üí   743‚Üí    if upsampling_ratio > 0:
   744‚Üí   744‚Üí        x_train, y_train = upsampling(x_train, y_train, upsampling_ratio, upsampling_mode)
   745‚Üí   745‚Üí        print(f"Upsampled training data to {x_train.shape[0]} samples.", flush=True)
   746‚Üí   746‚Üí
   747‚Üí   747‚Üí    # Apply mixup to training data
   748‚Üí   748‚Üí    if train_with_mixup and not cfg.BINARY_CLASSIFICATION:
   749‚Üí   749‚Üí        x_train, y_train = mixup(x_train, y_train)
   750‚Üí   750‚Üí
   751‚Üí   751‚Üí    # Apply label smoothing
   752‚Üí   752‚Üí    if train_with_label_smoothing and not cfg.BINARY_CLASSIFICATION:
   753‚Üí   753‚Üí        y_train = label_smoothing(y_train)
   754‚Üí   754‚Üí
   755‚Üí   755‚Üí    # Early stopping with patience depending on dataset size
   756‚Üí   756‚Üí    patience = min(10, max(5, int(epochs / 10)))
   757‚Üí   757‚Üí    min_delta = 0.001
   758‚Üí   758‚Üí
   759‚Üí   759‚Üí    callbacks = [
   760‚Üí   760‚Üí        # EarlyStopping with restore_best_weights
   761‚Üí   761‚Üí        keras.callbacks.EarlyStopping(
   762‚Üí   762‚Üí            monitor="val_loss",
   763‚Üí   763‚Üí            mode="min",
   764‚Üí   764‚Üí            patience=patience,
   765‚Üí   765‚Üí            verbose=1,
   766‚Üí   766‚Üí            min_delta=min_delta,
   767‚Üí   767‚Üí            restore_best_weights=True,
   768‚Üí   768‚Üí        ),
   769‚Üí   769‚Üí        # Function callback for progress tracking
   770‚Üí   770‚Üí        FunctionCallback(on_epoch_end=on_epoch_end),
   771‚Üí   771‚Üí    ]
   772‚Üí   772‚Üí

   775‚Üí   775‚Üí

   777‚Üí   777‚Üí        if epoch < warmup_epochs:
   778‚Üí   778‚Üí            # Linear warmup
   779‚Üí   779‚Üí            return learning_rate * (epoch + 1) / warmup_epochs
   780‚Üí   780‚Üí
   781‚Üí   781‚Üí        # Cosine decay
   782‚Üí   782‚Üí        progress = (epoch - warmup_epochs) / (epochs - warmup_epochs)
   783‚Üí   783‚Üí        return learning_rate * (0.1 + 0.9 * (1 + np.cos(np.pi * progress)) / 2)
   784‚Üí   784‚Üí
   785‚Üí   785‚Üí    # Add LR scheduler callback
   786‚Üí   786‚Üí    callbacks.append(keras.callbacks.LearningRateScheduler(lr_schedule))
   787‚Üí   787‚Üí
   788‚Üí   788‚Üí    def _focal_loss(y_true, y_pred):

   790‚Üí   790‚Üí
   791‚Üí   791‚Üí    # Choose the loss function based on config
   792‚Üí   792‚Üí    loss_function = _focal_loss if train_with_focal_loss else custom_loss
   793‚Üí   793‚Üí
   794‚Üí   794‚Üí    # Compile model with appropriate metrics for classification task
   795‚Üí   795‚Üí    classifier.compile(
   796‚Üí   796‚Üí        optimizer=keras.optimizers.Adam(learning_rate=learning_rate),
   797‚Üí   797‚Üí        loss=loss_function,
   798‚Üí   798‚Üí        metrics=[
   799‚Üí   799‚Üí            keras.metrics.AUC(
   800‚Üí   800‚Üí                curve="PR",
   801‚Üí   801‚Üí                multi_label=cfg.MULTI_LABEL,
   802‚Üí   802‚Üí                name="AUPRC",
   803‚Üí   803‚Üí                num_labels=y_train.shape[1] if cfg.MULTI_LABEL else None,
   804‚Üí   804‚Üí                from_logits=True,
   805‚Üí   805‚Üí            ),
   806‚Üí   806‚Üí            keras.metrics.AUC(
   807‚Üí   807‚Üí                curve="ROC",
   808‚Üí   808‚Üí                multi_label=cfg.MULTI_LABEL,
   809‚Üí   809‚Üí                name="AUROC",
   810‚Üí   810‚Üí                num_labels=y_train.shape[1] if cfg.MULTI_LABEL else None,
   811‚Üí   811‚Üí                from_logits=True,
   812‚Üí   812‚Üí            ),
   813‚Üí   813‚Üí        ],
   814‚Üí   814‚Üí    )
   815‚Üí   815‚Üí
   816‚Üí   816‚Üí    # Train model
   817‚Üí   817‚Üí    history = classifier.fit(x_train, y_train, epochs=epochs, batch_size=batch_size, validation_data=(x_val, y_val), callbacks=callbacks)
   818‚Üí   818‚Üí
   819‚Üí   819‚Üí    return classifier, history

  1021‚Üí  1021‚Üídef focal_loss(y_true, y_pred, gamma=2.0, alpha=0.25, epsilon=1e-7):
  1022‚Üí  1022‚Üí    """
  1023‚Üí  1023‚Üí    Focal loss for better handling of class imbalance.
  1024‚Üí  1024‚Üí
  1025‚Üí  1025‚Üí    This loss function gives more weight to hard examples and down-weights easy examples.
  1026‚Üí  1026‚Üí    Particularly helpful for imbalanced datasets where some classes have few samples.
  1027‚Üí  1027‚Üí
  1028‚Üí  1028‚Üí    Args:
  1029‚Üí  1029‚Üí        y_true: Ground truth labels.
  1030‚Üí  1030‚Üí        y_pred: Predicted probabilities.
  1031‚Üí  1031‚Üí        gamma: Focusing parameter. Higher values mean more focus on hard examples.
  1032‚Üí  1032‚Üí        alpha: Balance parameter. Controls weight of positive vs negative examples.
  1033‚Üí  1033‚Üí        epsilon: Small constant to prevent log(0).
  1034‚Üí  1034‚Üí
  1035‚Üí  1035‚Üí    Returns:
  1036‚Üí  1036‚Üí        Focal loss value.
  1037‚Üí  1037‚Üí    """
  1038‚Üí  1038‚Üí    # Apply sigmoid if not already applied
  1039‚Üí  1039‚Üí    y_pred = tf.clip_by_value(y_pred, epsilon, 1.0 - epsilon)
  1040‚Üí  1040‚Üí
  1041‚Üí  1041‚Üí    # Calculate cross entropy
  1042‚Üí  1042‚Üí    cross_entropy = -y_true * tf.math.log(y_pred) - (1 - y_true) * tf.math.log(1 - y_pred)
  1043‚Üí  1043‚Üí
  1044‚Üí  1044‚Üí    # Calculate focal weight
  1045‚Üí  1045‚Üí    p_t = y_true * y_pred + (1 - y_true) * (1 - y_pred)
  1046‚Üí  1046‚Üí    focal_weight = tf.pow(1 - p_t, gamma)
  1047‚Üí  1047‚Üí
  1048‚Üí  1048‚Üí    # Apply alpha balancing
  1049‚Üí  1049‚Üí    alpha_factor = y_true * alpha + (1 - y_true) * (1 - alpha)
  1050‚Üí  1050‚Üí
  1051‚Üí  1051‚Üí    # Calculate focal loss
  1052‚Üí  1052‚Üí    focal_loss = alpha_factor * focal_weight * cross_entropy
  1053‚Üí  1053‚Üí
  1054‚Üí  1054‚Üí    # Sum over all classes
  1055‚Üí  1055‚Üí    return tf.reduce_sum(focal_loss, axis=-1)

  1058‚Üí  1058‚Üídef custom_loss(y_true, y_pred, epsilon=1e-7):
  1059‚Üí  1059‚Üí    # Calculate loss for positive labels with epsilon
  1060‚Üí  1060‚Üí    positive_loss = -tf.reduce_sum(y_true * tf.math.log(tf.clip_by_value(y_pred, epsilon, 1.0 - epsilon)), axis=-1)
  1061‚Üí  1061‚Üí
  1062‚Üí  1062‚Üí    # Calculate loss for negative labels with epsilon
  1063‚Üí  1063‚Üí    negative_loss = -tf.reduce_sum((1 - y_true) * tf.math.log(tf.clip_by_value(1 - y_pred, epsilon, 1.0 - epsilon)), axis=-1)
  1064‚Üí  1064‚Üí
  1065‚Üí  1065‚Üí    # Combine both loss terms
  1066‚Üí  1066‚Üí    return positive_loss + negative_loss

  1069‚Üí  1069‚Üídef flat_sigmoid(x, sensitivity=-1, bias=1.0):
  1070‚Üí  1070‚Üí    """
  1071‚Üí  1071‚Üí    Applies a flat sigmoid function to the input array with a bias shift.
  1072‚Üí  1072‚Üí
  1073‚Üí  1073‚Üí    The flat sigmoid function is defined as:
  1074‚Üí  1074‚Üí        f(x) = 1 / (1 + exp(sensitivity * clip(x + bias, -20, 20)))
  1075‚Üí  1075‚Üí
  1076‚Üí  1076‚Üí    We transform the bias parameter to a range of [-100, 100] with the formula:
  1077‚Üí  1077‚Üí        transformed_bias = (bias - 1.0) * 10.0
  1078‚Üí  1078‚Üí
  1079‚Üí  1079‚Üí    Thus, higher bias values will shift the sigmoid function to the right on the x-axis, making it more "sensitive".
  1080‚Üí  1080‚Üí
  1081‚Üí  1081‚Üí    Note: Not sure why we are clipping, must be for numerical stability somewhere else in the code.
  1082‚Üí  1082‚Üí
  1083‚Üí  1083‚Üí    Args:
  1084‚Üí  1084‚Üí        x (array-like): Input data.
  1085‚Üí  1085‚Üí        sensitivity (float, optional): Sensitivity parameter for the sigmoid function. Default is -1.
  1086‚Üí  1086‚Üí        bias (float, optional): Bias parameter to shift the sigmoid function on the x-axis. Must be in the range [0.01, 1.99]. Default is 1.0.
  1087‚Üí  1087‚Üí
  1088‚Üí  1088‚Üí    Returns:
  1089‚Üí  1089‚Üí        numpy.ndarray: Transformed data after applying the flat sigmoid function.
  1090‚Üí  1090‚Üí    """
  1091‚Üí  1091‚Üí
  1092‚Üí  1092‚Üí    transformed_bias = (bias - 1.0) * 10.0
  1093‚Üí  1093‚Üí
  1094‚Üí  1094‚Üí    return 1 / (1.0 + np.exp(sensitivity * np.clip(x + transformed_bias, -20, 20)))
